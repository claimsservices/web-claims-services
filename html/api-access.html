คู่มือการใช้งาน API (API Documentation)

Base URL: https://be-claims-service.onrender.com/api

Authentication: ส่ง Token ใน Header Authorization: Bearer <YOUR_TOKEN>

Endpoint: POST /orders/inquiry

คำอธิบาย: ใช้สำหรับค้นหารายการงานเคลมทั้งหมดที่เกี่ยวข้องกับบริษัทประกันของท่าน สามารถกรองตามสถานะงานได้

Request Body:

{
  "order_status": "รับเรื่องแล้ว", // (Optional) กรองตามสถานะ
  "id": "ST-INS-",                // (Optional) ค้นหาจากรหัสงาน
  "car_registration": "กข-1234"   // (Optional) ค้นหาจากทะเบียนรถ
}

Endpoint: POST /order-detail/inquiry

คำอธิบาย: ดึงข้อมูลรายละเอียดทั้งหมดของงาน 1 รายการ รวมถึงข้อมูลรถ, รูปภาพความเสียหาย, และประวัติการอัปเดตสถานะ

Request Body:

{
  "order_id": "ST-INS-260517-0001" // ID ของงานที่ต้องการดูรายละเอียด
}

Endpoint: POST /orders/create

คำอธิบาย: ใช้สำหรับสร้างงานเคลมใหม่เข้าระบบโดยตรง (บริษัทประกันจะถูกล็อกตามสิทธิ์ของท่านอัตโนมัติ)

Request Body (ตัวอย่าง):

{
  "car_registration": "กข-1234",
  "location": "กรุงเทพมหานคร",
  "order_type": "เคลมสด",
  "appointment_date": "2026-05-18T10:00:00+07:00",
  "tell_1": "0812345678",
  "c_name": "ชื่อผู้แจ้งเหตุ"
}

Endpoint: PUT /order-status/update/{order_id}

คำอธิบาย: ใช้สำหรับเปลี่ยนสถานะของงาน เช่น เมื่อบริษัทประกันตรวจสอบและอนุมัติงานแล้ว (ผ่าน, ไม่ผ่าน, แก้ไข, Pre-Approved)

Request Body:

{
  "order_status": "ผ่าน", // สถานะใหม่ที่ต้องการเปลี่ยน
  "updated_by": "ชื่อผู้ตรวจสอบ"
}

Endpoint: POST /upload/proxy-download

คำอธิบาย: ใช้สำหรับดาวน์โหลดไฟล์รูปภาพจากระบบ โดยส่ง URL ของรูปภาพที่ได้จาก Order Details มาให้ API ประมวลผล

Request Body:

{
  "imageUrl": "https://res.cloudinary.com/..." // URL ของรูปภาพ
}
# 1. สร้างงานเคลมใหม่
curl -X POST https://be-claims-service.onrender.com/api/orders/create \
  -H "Authorization: Bearer YOUR_TOKEN" \
  -H "Content-Type: application/json" \
  -d '{"car_registration": "กข-1234", "location": "กทม", "order_type": "เคลมสด"}'

# 2. ค้นหาข้อมูลงาน
curl -X POST https://be-claims-service.onrender.com/api/orders/inquiry \
  -H "Authorization: Bearer YOUR_TOKEN" \
  -H "Content-Type: application/json" \
  -d '{"order_status": "รับเรื่องแล้ว"}'

# 3. อัปเดตสถานะงาน
curl -X PUT https://be-claims-service.onrender.com/api/order-status/update/ST-INS-260517-0001 \
  -H "Authorization: Bearer YOUR_TOKEN" \
  -H "Content-Type: application/json" \
  -d '{"order_status": "ผ่าน", "updated_by": "User Name"}'
// ตัวอย่างการส่งข้อมูลสร้างงานใหม่
const createOrder = async (orderData) => {
  const response = await fetch('https://be-claims-service.onrender.com/api/orders/create', {
    method: 'POST',
    headers: {
      'Authorization': 'Bearer YOUR_TOKEN',
      'Content-Type': 'application/json'
    },
    body: JSON.stringify(orderData)
  });
  return await response.json();
};
